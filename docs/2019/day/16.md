  let line = space + Array(header.length - space.length).fill('-').join('');
  let grid = [];

    let row = [];
    let pattern = fillPattern(y + 1);
    for (let x = 0; x < input.length; x++) {
      let char = '    ';
      let i = x % pattern.length;
      let scalar = pattern[i];

      if (y >= mid) {
        if (scalar === 1) {
          char = ` *${input[x]}*`;
        } else if (scalar === -1) {
          char = `*-${input[x]}*`;
        }
      } else {
        if (scalar === 1) {
          char = `  ${input[x]} `;
        } else if (scalar === -1) {
          char = ` -${input[x]} `;
        }
      }

      row.push(char);
    }

    grid.push(`${(y+1).toString().padStart(pad)} | ${row.join(' | ')} |`);
  }

  let output = [header, line, ...grid].join('\n');
  return output;
}
